using MediatR;
using Microsoft.EntityFrameworkCore;

namespace MyApp.Features.Products.GetProductList;

public class GetProductListHandler
    : IRequestHandler<GetProductListQuery, GetProductListResult>
{
    private readonly AppDbContext _db;

    public GetProductListHandler(AppDbContext db)
    {
        _db = db;
    }

    public async Task<GetProductListResult> Handle(
        GetProductListQuery request,
        CancellationToken cancellationToken)
    {
        var query = _db.Products.AsQueryable();

        // í•„í„°ë§
        if (!string.IsNullOrEmpty(request.Category))
        {
            query = query.Where(p => p.Category == request.Category);
        }

        // ì´ ê°œìˆ˜
        var totalCount = await query.CountAsync(cancellationToken);
        var totalPages = (int)Math.Ceiling(totalCount / (double)request.PageSize);

        // í˜ì´ì§• ë° ë§¤í•‘ (Entity â†’ DTO ì§ì ‘ ë³€í™˜)
        var products = await query
            .OrderBy(p => p.Name)
            .Skip((request.Page - 1) * request.PageSize)
            .Take(request.PageSize)
            .Select(p => new ProductListItem(
                p.Id,
                p.Name,
                p.Price,
                p.Category,
                p.Stock
            ))
            .ToListAsync(cancellationToken);

        return new GetProductListResult(products, totalCount, totalPages);
    }
}
