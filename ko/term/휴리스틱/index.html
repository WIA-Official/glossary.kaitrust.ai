import heapq

# 1. A* ì•Œê³ ë¦¬ì¦˜ - íœ´ë¦¬ìŠ¤í‹± í•¨ìˆ˜ë¥¼ ì‚¬ìš©í•œ ìµœë‹¨ ê²½ë¡œ
def a_star_search(
    start: tuple,
    goal: tuple,
    heuristic: Callable[[tuple, tuple], float],
    get_neighbors: Callable[[tuple], list],
    cost: Callable[[tuple, tuple], float] = lambda a, b: 1
):
    """A* íƒìƒ‰ ì•Œê³ ë¦¬ì¦˜ - h(n) íœ´ë¦¬ìŠ¤í‹±ìœ¼ë¡œ íƒìƒ‰ íš¨ìœ¨í™”"""
    open_set = [(0, start)]  # (f_score, node)
    came_from = {}
    g_score = {start: 0}
    f_score = {start: heuristic(start, goal)}

    while open_set:
        _, current = heapq.heappop(open_set)

        if current == goal:
            # ê²½ë¡œ ë³µì›
            path = [current]
            while current in came_from:
                current = came_from[current]
                path.append(current)
            return path[::-1]

        for neighbor in get_neighbors(current):
            tentative_g = g_score[current] + cost(current, neighbor)

            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                f_score[neighbor] = tentative_g + heuristic(neighbor, goal)
                heapq.heappush(open_set, (f_score[neighbor], neighbor))

    return None  # ê²½ë¡œ ì—†ìŒ

def manhattan_distance(a: tuple, b: tuple) -> float:
    """ë§¨í•´íŠ¼ ê±°ë¦¬ íœ´ë¦¬ìŠ¤í‹± (ê·¸ë¦¬ë“œìš©)"""
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

def euclidean_distance(a: tuple, b: tuple) -> float:
    """ìœ í´ë¦¬ë“œ ê±°ë¦¬ íœ´ë¦¬ìŠ¤í‹±"""
    return math.sqrt((a[0] - b[0])**2 + (a[1] - b[1])**2)

# 2. íƒìš• ì•Œê³ ë¦¬ì¦˜ - ì‘ì—… ìŠ¤ì¼€ì¤„ë§ íœ´ë¦¬ìŠ¤í‹±
def greedy_job_scheduling(jobs: list[tuple[int, int, int]]) -> list[int]:
    """
    ì‘ì—… ìŠ¤ì¼€ì¤„ë§: (ì‹œì‘ì‹œê°„, ì¢…ë£Œì‹œê°„, ê°€ì¹˜)
    íœ´ë¦¬ìŠ¤í‹±: ê°€ì¥ ë¹¨ë¦¬ ëë‚˜ëŠ” ì‘ì—… ë¨¼ì € (EDF)
    """
    # ì¢…ë£Œ ì‹œê°„ ê¸°ì¤€ ì •ë ¬ (EDF íœ´ë¦¬ìŠ¤í‹±)
    sorted_jobs = sorted(enumerate(jobs), key=lambda x: x[1][1])

    selected = []
    last_end = 0

    for idx, (start, end, value) in sorted_jobs:
        if start >= last_end:
            selected.append(idx)
            last_end = end

    return selected

# 3. ì‹œë®¬ë ˆì´í‹°ë“œ ì–´ë‹ë§ - ë©”íƒ€íœ´ë¦¬ìŠ¤í‹±
def simulated_annealing(
    initial_state,
    objective: Callable,
    neighbor: Callable,
    initial_temp: float = 1000,
    cooling_rate: float = 0.995,
    min_temp: float = 1
):
    """
    ì‹œë®¬ë ˆì´í‹°ë“œ ì–´ë‹ë§ - ì§€ì—­ ìµœì í•´ íƒˆì¶œ íœ´ë¦¬ìŠ¤í‹±
    """
    current = initial_state
    current_score = objective(current)
    best = current
    best_score = current_score
    temperature = initial_temp

    while temperature > min_temp:
        # ì´ì›ƒ í•´ ìƒì„±
        candidate = neighbor(current)
        candidate_score = objective(candidate)

        # ìˆ˜ë½ í™•ë¥  ê³„ì‚° (íœ´ë¦¬ìŠ¤í‹±)
        delta = candidate_score - current_score

        if delta < 0 or random.random() < math.exp(-delta / temperature):
            current = candidate
            current_score = candidate_score

            if current_score < best_score:
                best = current
                best_score = current_score

        temperature *= cooling_rate

    return best, best_score

def beam_search_decode(
    model_predict: Callable,  # ë‹¤ìŒ í† í° í™•ë¥  ì˜ˆì¸¡ í•¨ìˆ˜
    start_token: int,
    end_token: int,
    beam_width: int = 5,  # íœ´ë¦¬ìŠ¤í‹±: íƒìƒ‰ ë„ˆë¹„
    max_length: int = 50
) -> list[int]:
    """ë¹” ì„œì¹˜ - í™•ë¥  ê¸°ë°˜ íƒìƒ‰ íœ´ë¦¬ìŠ¤í‹±"""
    # (ëˆ„ì  log í™•ë¥ , ì‹œí€€ìŠ¤)
    beams = [(0.0, [start_token])]

    for _ in range(max_length):
        candidates = []

        for score, sequence in beams:
            if sequence[-1] == end_token:
                candidates.append((score, sequence))
                continue

            # ë‹¤ìŒ í† í° í™•ë¥  ì˜ˆì¸¡ (ì‹¤ì œë¡œëŠ” ëª¨ë¸ í˜¸ì¶œ)
            next_probs = model_predict(sequence)

            # ìƒìœ„ kê°œë§Œ ê³ ë ¤ (íœ´ë¦¬ìŠ¤í‹±)
            top_k = sorted(enumerate(next_probs), key=lambda x: -x[1])[:beam_width]

            for token, prob in top_k:
                new_score = score + math.log(prob + 1e-10)
                candidates.append((new_score, sequence + [token]))

        # ìƒìœ„ ë¹”ë§Œ ìœ ì§€ (íœ´ë¦¬ìŠ¤í‹±)
        beams = sorted(candidates, key=lambda x: -x[0])[:beam_width]

        if all(seq[-1] == end_token for _, seq in beams):
            break

    return beams[0][1]  # ìµœê³  ì ìˆ˜ ì‹œí€€ìŠ¤

class AnomalyDetectionHeuristics:
    """í†µê³„ ê¸°ë°˜ ì´ìƒ íƒì§€ íœ´ë¦¬ìŠ¤í‹±"""

    @staticmethod
    def three_sigma_rule(data: list[float], value: float) -> bool:
        """3-ì‹œê·¸ë§ˆ ê·œì¹™: í‰ê· ì—ì„œ 3 í‘œì¤€í¸ì°¨ ì´ìƒì´ë©´ ì´ìƒ"""
        mean = sum(data) / len(data)
        std = (sum((x - mean)**2 for x in data) / len(data)) ** 0.5
        return abs(value - mean) > 3 * std

    @staticmethod
    def iqr_rule(data: list[float], value: float) -> bool:
        """IQR ê·œì¹™: Q1 - 1.5*IQR ë¯¸ë§Œ ë˜ëŠ” Q3 + 1.5*IQR ì´ˆê³¼ë©´ ì´ìƒ"""
        sorted_data = sorted(data)
        n = len(sorted_data)
        q1 = sorted_data[n // 4]
        q3 = sorted_data[3 * n // 4]
        iqr = q3 - q1
        return value < q1 - 1.5 * iqr or value > q3 + 1.5 * iqr

    @staticmethod
    def sudden_change(current: float, previous: float, threshold: float = 0.5) -> bool:
        """ê¸‰ê²©í•œ ë³€í™” ê°ì§€: ì´ì „ ê°’ ëŒ€ë¹„ threshold% ì´ìƒ ë³€í™”"""
        if previous == 0:
            return abs(current) > threshold
        return abs((current - previous) / previous) > threshold
