class ReplicaAwareDBPool:
    """
    Primary-Replica êµ¬ì¡°ì˜ DB ì—°ê²° í’€
    ì“°ê¸°: Primary, ì½ê¸°: Replica(s)
    """

    def __init__(self, primary_config: dict, replica_configs: list):
        # Primary ì—°ê²° í’€ (ì“°ê¸° ì „ìš©)
        self.primary_pool = pool.ThreadedConnectionPool(
            minconn=2,
            maxconn=10,
            host=primary_config['host'],
            database=primary_config['database'],
            user=primary_config['user'],
            password=primary_config['password']
        )

        # Replica ì—°ê²° í’€ë“¤ (ì½ê¸° ì „ìš©)
        self.replica_pools = []
        for config in replica_configs:
            replica_pool = pool.ThreadedConnectionPool(
                minconn=2,
                maxconn=20,  # ì½ê¸°ëŠ” ë” ë§ì€ ì—°ê²°
                host=config['host'],
                database=config['database'],
                user=config['user'],
                password=config['password']
            )
            self.replica_pools.append(replica_pool)

    @contextmanager
    def get_primary_connection(self):
        """ì“°ê¸° ì‘ì—…ìš© Primary ì—°ê²°"""
        conn = self.primary_pool.getconn()
        try:
            yield conn
            conn.commit()
        except Exception as e:
            conn.rollback()
            raise e
        finally:
            self.primary_pool.putconn(conn)

    @contextmanager
    def get_replica_connection(self):
        """
        ì½ê¸° ì‘ì—…ìš© Replica ì—°ê²°
        ë¼ìš´ë“œ ë¡œë¹ˆ ë˜ëŠ” ëœë¤ìœ¼ë¡œ Replica ì„ íƒ
        """
        if not self.replica_pools:
            # Replicaê°€ ì—†ìœ¼ë©´ Primary ì‚¬ìš©
            with self.get_primary_connection() as conn:
                yield conn
                return

        # ëœë¤í•˜ê²Œ Replica ì„ íƒ (ë¡œë“œ ë°¸ëŸ°ì‹±)
        replica_pool = random.choice(self.replica_pools)
        conn = replica_pool.getconn()
        try:
            yield conn
        finally:
            replica_pool.putconn(conn)

    ì§€ì—°ì´ í¬ë©´ Replica ì½ê¸°ì— ì£¼ì˜ í•„ìš”
    """
    with replica_conn.cursor() as cur:
        cur.execute("""
            SELECT EXTRACT(EPOCH FROM (NOW() - pg_last_xact_replay_timestamp()))
            AS lag_seconds
        """)
        result = cur.fetchone()
        lag = result[0] if result[0] else 0
        print(f"â±ï¸ Replication Lag: {lag:.2f}ì´ˆ")
        return lag

            where: { id: userId }

    }

    /**
     * íŠ¸ëœì­ì…˜ ë‚´ì—ì„œ ì½ê¸° í›„ ì“°ê¸°
     * íŠ¸ëœì­ì…˜ì€ í•­ìƒ Primaryì—ì„œ ì‹¤í–‰

    async transferBalance(fromUserId, toUserId, amount) {
        const transaction = await sequelize.transaction();

        try {
            // íŠ¸ëœì­ì…˜ ë‚´ ëª¨ë“  ì¿¼ë¦¬ëŠ” Primaryì—ì„œ ì‹¤í–‰

                throw new Error('ì”ì•¡ ë¶€ì¡±');

                { balance: fromUser.balance - amount },
                { transaction }

                { transaction }
            );

            throw error;
        }
    }

async function checkReplicationLag() {
    const [results] = await sequelize.query(`
        SELECT
            client_addr,
            state,
            sent_lsn,
            write_lsn,
            flush_lsn,
            replay_lsn,
            EXTRACT(EPOCH FROM (now() - reply_time)) AS lag_seconds
        FROM pg_stat_replication
    `, { type: Sequelize.QueryTypes.SELECT, useMaster: true });

    results.forEach(replica => {
        console.log(`ğŸ“Š Replica ${replica.client_addr}: ${replica.lag_seconds}ì´ˆ ì§€ì—°`);
    });

    return results;
}
