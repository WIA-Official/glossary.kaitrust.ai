import gymnasium as gym
import numpy as np

# CartPole í™˜ê²½ ìƒì„±
env = gym.make("CartPole-v1")

# Q-Table ì´ˆê¸°í™” (ìƒíƒœë¥¼ ì´ì‚°í™”í•˜ì—¬ ì‚¬ìš©)
def discretize_state(state):
    bins = [
        np.linspace(-4.8, 4.8, 10),   # cart position
        np.linspace(-4, 4, 10),        # cart velocity
        np.linspace(-0.42, 0.42, 10),  # pole angle
        np.linspace(-4, 4, 10)         # pole velocity
    ]
    indices = []
    for i, val in enumerate(state):
        indices.append(np.digitize(val, bins[i]) - 1)
    return tuple(indices)

# Q-Table ì´ˆê¸°í™”
q_table = np.zeros((10, 10, 10, 10, env.action_space.n))

# í•˜ì´í¼íŒŒë¼ë¯¸í„°
learning_rate = 0.1
discount_factor = 0.99
epsilon = 1.0
epsilon_decay = 0.995

# í•™ìŠµ ë£¨í”„
for episode in range(1000):
    state, _ = env.reset()
    state = discretize_state(state)
    total_reward = 0

    while True:
        # Epsilon-greedy í–‰ë™ ì„ íƒ
        if np.random.random() < epsilon:
            action = env.action_space.sample()
        else:
            action = np.argmax(q_table[state])

        # í™˜ê²½ì—ì„œ í–‰ë™ ìˆ˜í–‰
        next_state, reward, terminated, truncated, _ = env.step(action)
        next_state = discretize_state(next_state)

        # Q-Value ì—…ë°ì´íŠ¸ (ë²¨ë§Œ ë°©ì •ì‹)
        best_next_q = np.max(q_table[next_state])
        q_table[state + (action,)] += learning_rate * (
            reward + discount_factor * best_next_q - q_table[state + (action,)]
        )

        state = next_state
        total_reward += reward

        if terminated or truncated:
            break

    epsilon *= epsilon_decay

    if episode % 100 == 0:
        print(f"Episode {episode}, Total Reward: {total_reward:.0f}")

env.close()
