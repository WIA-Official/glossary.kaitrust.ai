class User(Base):
    __tablename__ = 'users'
    id = Column(Integer, primary_key=True)
    name = Column(String(100))
    email = Column(String(255))
    status = Column(String(20))
    created_at = Column(DateTime, default=datetime.utcnow)
    orders = relationship("Order", back_populates="user")

class Order(Base):
    __tablename__ = 'orders'
    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey('users.id'))
    total_amount = Column(Numeric(10, 2))
    status = Column(String(20))
    created_at = Column(DateTime)
    user = relationship("User", back_populates="orders")


class QueryExamples:
    def __init__(self, engine):
        self.engine = engine

    def basic_select(self):
        """ê¸°ë³¸ SELECT ì¿¼ë¦¬"""
        with Session(self.engine) as session:
            # ORM ìŠ¤íƒ€ì¼
            users = session.query(User).filter(
                User.status == 'active'

            # 2.0 ìŠ¤íƒ€ì¼ (ê¶Œì¥)
            stmt = select(User).where(
                User.status == 'active'

            return users

    def complex_filter(self):
        """ë³µì¡í•œ ì¡°ê±´ í•„í„°"""
        with Session(self.engine) as session:
            one_month_ago = datetime.utcnow() - timedelta(days=30)

            # AND, OR ì¡°ê±´
            stmt = select(User).where(
                and_(
                    User.status == 'active',
                    or_(
                        User.email.like('%@gmail.com'),
                        User.created_at >= one_month_ago
                    )
                )
            )
            return session.scalars(stmt).all()

    def join_query(self):
        """JOIN ì¿¼ë¦¬"""
        with Session(self.engine) as session:
            # ì‚¬ìš©ìì™€ ì£¼ë¬¸ ì¡°ì¸
            stmt = (
                select(User.name, Order.total_amount, Order.created_at)
                .join(Order, User.id == Order.user_id)
                .where(Order.status == 'completed')
                .order_by(Order.created_at.desc())
            )
            results = session.execute(stmt).all()

            # Eager Loadingìœ¼ë¡œ N+1 ë°©ì§€
            from sqlalchemy.orm import selectinload
            stmt = select(User).options(
                selectinload(User.orders)
            ).where(User.status == 'active')
            users_with_orders = session.scalars(stmt).all()

            return results

    def aggregate_query(self):
        """ì§‘ê³„ ì¿¼ë¦¬"""
        with Session(self.engine) as session:
            # ì›”ë³„ ë§¤ì¶œ í†µê³„
            stmt = (
                select(
                    func.date_trunc('month', Order.created_at).label('month'),
                    func.count(Order.id).label('order_count'),
                    func.sum(Order.total_amount).label('revenue'),
                    func.avg(Order.total_amount).label('avg_order')
                )
                .where(Order.status == 'completed')
                .group_by(func.date_trunc('month', Order.created_at))
                .having(func.sum(Order.total_amount) > 1000000)
                .order_by(text('month DESC'))
            )
            return session.execute(stmt).all()

    def subquery_example(self):
        """ì„œë¸Œì¿¼ë¦¬ ì˜ˆì œ"""

            stmt = select(User).where(
                User.id.in_(select(order_subquery.c.user_id))

            return session.scalars(stmt).all()

    def raw_sql(self):

                LEFT JOIN orders o ON u.id = o.user_id
                WHERE u.status = :status
                GROUP BY u.id, u.name
                HAVING COUNT(o.id) > :min_orders
                ORDER BY total_spent DESC
            """), {"status": "active", "min_orders": 5})

    def pagination(self, page: int = 1, per_page: int = 20):
        """í˜ì´ì§€ë„¤ì´ì…˜"""

const prisma = new PrismaClient({
    log: ['query'], // ì‹¤í–‰ë˜ëŠ” SQL í™•ì¸

class QueryExamples {
    /**

    async basicSelect() {

    /**

    async complexFilter() {

        // AND, OR ì¡°ê±´
        const users = await prisma.user.findMany({
            where: {
                AND: [
                    { status: 'active' },
                    {
                        OR: [
                            { email: { contains: '@gmail.com' } },
                            { createdAt: { gte: oneMonthAgo } },
                        ],
                    },
                ],
            },
        });

        // NOT ì¡°ê±´
        const notDeleted = await prisma.user.findMany({
            where: {
                NOT: { status: 'deleted' },
            },
        });

    }

    /**

    async joinQuery() {
        // ì‚¬ìš©ìì™€ ì£¼ë¬¸ í•¨ê»˜ ì¡°íšŒ

            where: { status: 'active' },

    async aggregateQuery() {

            by: ['category'],

    async rawQuery() {

    /**

    async pagination(page = 1, perPage = 20) {

        // íŠ¸ëœì­ì…˜ìœ¼ë¡œ ë°ì´í„°ì™€ ì¹´ìš´íŠ¸ ë™ì‹œ ì¡°íšŒ
        const [users, total] = await prisma.$transaction([
            prisma.user.findMany({
                where: { status: 'active' },
                orderBy: { id: 'asc' },
                skip,
                take: perPage,
            }),
            prisma.user.count({
                where: { status: 'active' },
            }),
        ]);

        return {
            data: users,
            meta: {
                total,
                page,
                perPage,
                totalPages: Math.ceil(total / perPage),

    /**
     * ì»¤ì„œ ê¸°ë°˜ í˜ì´ì§€ë„¤ì´ì…˜ (ëŒ€ìš©ëŸ‰ ìµœì í™”)

    async cursorPagination(cursor?: number, limit = 20) {
        const users = await prisma.user.findMany({
            take: limit + 1, // ë‹¤ìŒ í˜ì´ì§€ ì¡´ì¬ ì—¬ë¶€ í™•ì¸
            ...(cursor && {
                cursor: { id: cursor },
                skip: 1, // ì»¤ì„œ ìì²´ëŠ” ì œì™¸
            }),
            orderBy: { id: 'asc' },
        });

        const hasNextPage = users.length > limit;
        const data = hasNextPage ? users.slice(0, -1) : users;
        const nextCursor = hasNextPage ? data[data.length - 1]?.id : null;

        return { data, nextCursor, hasNextPage };
    }

    /**
     * íŠ¸ëœì­ì…˜ ì¿¼ë¦¬
     */
    async transactionQuery(fromUserId: number, toUserId: number, amount: number) {
        return prisma.$transaction(async (tx) => {
            // 1. ì¶œê¸ˆ
            const sender = await tx.wallet.update({
                where: { userId: fromUserId },
                data: { balance: { decrement: amount } },
            });

            // ì”ì•¡ ì²´í¬
            if (sender.balance < 0) {
                throw new Error('ì”ì•¡ ë¶€ì¡±');
            }

            // 2. ì…ê¸ˆ
            await tx.wallet.update({
                where: { userId: toUserId },
                data: { balance: { increment: amount } },
            });

            // 3. ê±°ë˜ ê¸°ë¡
            return tx.transaction.create({
                data: { fromUserId, toUserId, amount, type: 'transfer' },
            });
        });
    }
}
