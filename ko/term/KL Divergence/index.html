import torch
import torch.nn.functional as F

# NumPyë¡œ KL Divergence ì§ì ‘ êµ¬í˜„
def kl_divergence_numpy(p, q, epsilon=1e-10):
    """KL(P || Q): Pê°€ ì‹¤ì œ ë¶„í¬, Qê°€ ê·¼ì‚¬ ë¶„í¬"""
    p = np.array(p) + epsilon

# ì˜ˆì‹œ: ì‹¤ì œ ë¶„í¬ vs ì˜ˆì¸¡ ë¶„í¬

# VAEì—ì„œì˜ KL Divergence (ì ì¬ ê³µê°„ ì •ê·œí™”)
def vae_kl_loss(mu, log_var):
    """q(z|x) ~ N(mu, sigma^2)ì™€ p(z) ~ N(0, 1) ì‚¬ì´ì˜ KL"""
    return -0.5 * torch.sum(1 + log_var - mu.pow(2) - log_var.exp())
