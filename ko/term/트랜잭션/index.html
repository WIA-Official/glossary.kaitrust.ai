const { Pool } = require('pg');

class TransactionManager {
    constructor(connectionString) {
        this.pool = new Pool({ connectionString });
    }

    /**
     * íŠ¸ëœì­ì…˜ ë˜í¼ - ìë™ ì»¤ë°‹/ë¡¤ë°±
     * @param {Function} callback - íŠ¸ëœì­ì…˜ ë‚´ì—ì„œ ì‹¤í–‰í•  í•¨ìˆ˜
     */
    async withTransaction(callback) {
        const client = await this.pool.connect();

        try {
            await client.query('BEGIN');
            const result = await callback(client);
            await client.query('COMMIT');
            console.log('âœ… íŠ¸ëœì­ì…˜ ì»¤ë°‹ë¨');
            return result;

        } catch (error) {
            await client.query('ROLLBACK');
            console.error(`âŒ íŠ¸ëœì­ì…˜ ë¡¤ë°±ë¨: ${error.message}`);
            throw error;

        } finally {
            client.release();

    }
}

class OrderService {
    constructor(txManager) {
        this.txManager = txManager;
    }

    /**
     * ì£¼ë¬¸ ìƒì„± - íŠ¸ëœì­ì…˜ìœ¼ë¡œ ì—¬ëŸ¬ ì‘ì—… ë¬¶ê¸°
     */
    async createOrder(userId, items) {
        return await this.txManager.withTransaction(async (client) => {
            // 1. ì¬ê³  í™•ì¸ ë° ì°¨ê°
            for (const item of items) {
                const { rows } = await client.query(
                    'SELECT stock FROM products WHERE id = $1 FOR UPDATE',
                    [item.productId]
                );

                if (rows.length === 0 || rows[0].stock < item.quantity) {
                    throw new Error(`ì¬ê³  ë¶€ì¡±: ìƒí’ˆ ${item.productId}`);
                }

                await client.query(
                    'UPDATE products SET stock = stock - $1 WHERE id = $2',
                    [item.quantity, item.productId]
                );
            }

            // 2. ì£¼ë¬¸ ìƒì„±
            const totalAmount = items.reduce(
                (sum, item) => sum + item.price * item.quantity, 0
            );

            const orderResult = await client.query(
                `INSERT INTO orders (user_id, total_amount, status, created_at)
                 VALUES ($1, $2, 'pending', NOW())
                 RETURNING id`,
                [userId, totalAmount]
            );
            const orderId = orderResult.rows[0].id;

            // 3. ì£¼ë¬¸ ìƒì„¸ ìƒì„±
            for (const item of items) {
                await client.query(
                    `INSERT INTO order_items (order_id, product_id, quantity, price)
                     VALUES ($1, $2, $3, $4)`,
                    [orderId, item.productId, item.quantity, item.price]
                );
            }

            // 4. í¬ì¸íŠ¸ ì ë¦½
            const points = Math.floor(totalAmount * 0.01);
            await client.query(
                'UPDATE users SET points = points + $1 WHERE id = $2',
                [points, userId]
            );

            return orderId;
        });
    }
}

async function executeWithRetry(txManager, operation, maxRetries = 3) {
    let lastError;

    for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
            return await txManager.withTransaction(operation);
        } catch (error) {
            lastError = error;

            // ì§ë ¬í™” ì‹¤íŒ¨ ë˜ëŠ” ë°ë“œë½ì¸ ê²½ìš°ë§Œ ì¬ì‹œë„
            if (error.code === '40001' || error.code === '40P01') {
                console.log(`ì¬ì‹œë„ ${attempt}/${maxRetries}...`);
                await new Promise(r => setTimeout(r, 100 * attempt));
            } else {
                throw error;
            }
        }
    }

    throw lastError;
}
