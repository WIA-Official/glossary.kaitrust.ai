class CSPMScanner:
    """í´ë¼ìš°ë“œ ë³´ì•ˆ íƒœì„¸ ìë™ ìŠ¤ìºë„ˆ"""

    def __init__(self):

        self.findings = []

    def scan_s3_buckets(self):
        """S3 ë²„í‚· ë³´ì•ˆ ì„¤ì • ê²€ì‚¬"""
        buckets = self.s3.list_buckets()['Buckets']

        for bucket in buckets:

            try:
                public_access = self.s3.get_public_access_block(Bucket=bucket_name)

                    config.get('BlockPublicAcls'),
                    config.get('BlockPublicPolicy'),
                    config.get('IgnorePublicAcls'),
                    config.get('RestrictPublicBuckets')

                    self.findings.append({
                        'severity': 'HIGH',
                        'resource': f's3://{bucket_name}',
                        'rule': 'S3_PUBLIC_ACCESS_BLOCK',

                    'severity': 'CRITICAL',
                    'resource': f's3://{bucket_name}',
                    'rule': 'S3_NO_PUBLIC_ACCESS_BLOCK',
                    'message': 'í¼ë¸”ë¦­ ì•¡ì„¸ìŠ¤ ì°¨ë‹¨ ì„¤ì •ì´ ì—†ìŒ'
                })

            # ì•”í˜¸í™” í™•ì¸

                self.findings.append({
                    'severity': 'HIGH',
                    'resource': f's3://{bucket_name}',
                    'rule': 'S3_ENCRYPTION_DISABLED',
                    'message': 'ì„œë²„ ì‚¬ì´ë“œ ì•”í˜¸í™”ê°€ ë¹„í™œì„±í™”ë¨'
                })

    def scan_security_groups(self):
        """ë³´ì•ˆ ê·¸ë£¹ ê³¼ë„í•œ ê°œë°© ê²€ì‚¬"""
        sgs = self.ec2.describe_security_groups()['SecurityGroups']

                                'message': f'ë¯¼ê°í•œ í¬íŠ¸ {port}ê°€ ì¸í„°ë„·ì— ê°œë°©ë¨'

    def scan_iam_policies(self):
        """IAM ê³¼ë‹¤ ê¶Œí•œ ê²€ì‚¬"""
        users = self.iam.list_users()['Users']

                        'rule': 'IAM_ADMIN_ACCESS',
                        'message': 'IAM ì‚¬ìš©ìì—ê²Œ AdministratorAccess ì§ì ‘ ë¶€ì—¬ë¨'

                'HIGH': len([f for f in self.findings if f['severity'] == 'HIGH']),
                'MEDIUM': len([f for f in self.findings if f['severity'] == 'MEDIUM'])
            },
            'findings': self.findings
