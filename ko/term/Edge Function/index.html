// 2. KV Store í™œìš© - ê¸€ë¡œë²Œ í‚¤-ê°’ ì €ì¥ì†Œ
async function handleKV(request, env) {
  const url = new URL(request.url);
  const key = url.searchParams.get('key');

  if (request.method === 'GET') {
    // KVì—ì„œ ì½ê¸° (ì „ ì„¸ê³„ ì—£ì§€ì— ìºì‹œë¨)
    const value = await env.MY_KV.get(key, { type: 'json' });

    if (!value) {
      return Response.json({ error: 'Key not found' }, { status: 404 });
    }

    return Response.json({
      key,
      value,
      source: 'edge-kv',
    });
  }

  if (request.method === 'PUT') {
    const body = await request.json();

    // KVì— ì“°ê¸° (60ì´ˆ í›„ ì „ ì„¸ê³„ ë™ê¸°í™”)

      metadata: { updatedAt: Date.now() },
    });

    return Response.json({ success: true, key });
  }

  return new Response('Method not allowed', { status: 405 });
}

// 1. Atomic Counter with Deno KV
async function handleCounter(request: Request): Promise {
  const key = ["counters", "visits"];

  if (request.method === "POST") {
    // Atomic increment - ë™ì‹œì„± ì•ˆì „
    const result = await kv.atomic()
      .sum(key, 1n) // BigIntë¡œ ì¹´ìš´íŠ¸
      .commit();

    if (!result.ok) {
      return Response.json({ error: "Conflict" }, { status: 409 });
    }
  }

  // í˜„ì¬ ê°’ ì½ê¸°
  const entry = await kv.get(key);
  const count = entry.value?.value ?? 0n;

  return Response.json({
    count: count.toString(),
    timestamp: new Date().toISOString(),
  });
}

// 2. ì‚¬ìš©ì ì„¸ì…˜ ê´€ë¦¬
async function handleUser(request: Request): Promise {
  const url = new URL(request.url);
  const userId = url.searchParams.get("id");

  if (!userId) {
    return Response.json({ error: "Missing user ID" }, { status: 400 });
  }

  const key = ["users", userId];

  if (request.method === "GET") {
    const entry = await kv.get(key);

    if (!entry.value) {
      return Response.json({ error: "User not found" }, { status: 404 });
    }

    return Response.json(entry.value);
  }

  if (request.method === "PUT") {
    const body = await request.json();

    await kv.set(key, {
      ...body,
      updatedAt: Date.now(),
    }, {
      expireIn: 24 * 60 * 60 * 1000, // 24ì‹œê°„ í›„ ë§Œë£Œ
    });

    return Response.json({ success: true });
  }

  return new Response("Method not allowed", { status: 405 });
}

// 3. Streaming Response - ì‹¤ì‹œê°„ ë°ì´í„°
function handleStream(): Response {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    async start(controller) {
      for (let i = 0; i < 5; i++) {
        const data = {
          event: "update",
          data: { count: i, time: Date.now() },
        };

        controller.enqueue(
          encoder.encode(`data: ${JSON.stringify(data)}\n\n`)
        );

        await new Promise((r) => setTimeout(r, 1000));
      }

      controller.close();
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
    },
  });
}
