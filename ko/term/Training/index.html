import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

# ê°„ë‹¨í•œ ì‹ ê²½ë§ ì •ì˜
class SimpleNN(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim):
        super().__init__()
        self.layers = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim, output_dim)
        )

    def forward(self, x):
        return self.layers(x)

# ë°ì´í„° ì¤€ë¹„ (ì˜ˆì‹œ)
X = torch.randn(1000, 10)  # 1000ê°œ ìƒ˜í”Œ, 10ê°œ íŠ¹ì„±
y = torch.randint(0, 2, (1000,))  # ì´ì§„ ë¶„ë¥˜

dataset = TensorDataset(X, y)
train_loader = DataLoader(dataset, batch_size=32, shuffle=True)

# ëª¨ë¸, ì†ì‹¤í•¨ìˆ˜, ì˜µí‹°ë§ˆì´ì € ì„¤ì •
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model = SimpleNN(input_dim=10, hidden_dim=64, output_dim=2).to(device)
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

# Training Loop
num_epochs = 10
for epoch in range(num_epochs):
    model.train()  # í›ˆë ¨ ëª¨ë“œ
    total_loss = 0
    correct = 0

    for batch_X, batch_y in train_loader:
        batch_X, batch_y = batch_X.to(device), batch_y.to(device)

        # Forward pass
        outputs = model(batch_X)
        loss = criterion(outputs, batch_y)

        # Backward pass
        optimizer.zero_grad()  # ê¸°ìš¸ê¸° ì´ˆê¸°í™”
        loss.backward()        # ì—­ì „íŒŒ
        optimizer.step()       # ê°€ì¤‘ì¹˜ ì—…ë°ì´íŠ¸

        total_loss += loss.item()
        correct += (outputs.argmax(dim=1) == batch_y).sum().item()

    accuracy = correct / len(dataset) * 100
    print(f"Epoch [{epoch+1}/{num_epochs}] "
          f"Loss: {total_loss/len(train_loader):.4f} "
          f"Accuracy: {accuracy:.2f}%")

# ëª¨ë¸ ì €ì¥
torch.save(model.state_dict(), "model.pth")
print("Training complete! Model saved.")
