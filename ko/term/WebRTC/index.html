const express = require('express');
const http = require('http');
const { Server } = require('socket.io');

const app = express();
const server = http.createServer(app);
const io = new Server(server, {
    cors: { origin: '*' }
});

const rooms = new Map();

io.on('connection', (socket) => {
    console.log('User connected:', socket.id);

    // ë°© ì°¸ê°€
    socket.on('join-room', (roomId) => {
        socket.join(roomId);

        if (!rooms.has(roomId)) {
            rooms.set(roomId, new Set());
        }
        rooms.get(roomId).add(socket.id);

        // ê¸°ì¡´ ì°¸ê°€ìë“¤ì—ê²Œ ì•Œë¦¼
        socket.to(roomId).emit('user-joined', socket.id);

        // ê¸°ì¡´ ì°¸ê°€ì ëª©ë¡ ì „ì†¡
        const existingUsers = [...rooms.get(roomId)].filter(id => id !== socket.id);
        socket.emit('existing-users', existingUsers);
    });

    // Offer ì „ë‹¬
    socket.on('offer', ({ targetId, sdp }) => {
        io.to(targetId).emit('offer', {
            senderId: socket.id,
            sdp
        });
    });

    // Answer ì „ë‹¬
    socket.on('answer', ({ targetId, sdp }) => {
        io.to(targetId).emit('answer', {
            senderId: socket.id,
            sdp
        });
    });

    // ICE Candidate ì „ë‹¬
    socket.on('ice-candidate', ({ targetId, candidate }) => {
        io.to(targetId).emit('ice-candidate', {
            senderId: socket.id,
            candidate
        });
    });

    // ì—°ê²° í•´ì œ
    socket.on('disconnect', () => {
        rooms.forEach((users, roomId) => {
            if (users.has(socket.id)) {
                users.delete(socket.id);
                socket.to(roomId).emit('user-left', socket.id);
            }
        });
    });
});

async function startScreenShare() {
    try {
        // í™”ë©´ ìº¡ì²˜ ìŠ¤íŠ¸ë¦¼ íšë“
        const screenStream = await navigator.mediaDevices.getDisplayMedia({
            video: {
                cursor: 'always',  // ë§ˆìš°ìŠ¤ ì»¤ì„œ í‘œì‹œ

        });

        // ê¸°ì¡´ ë¹„ë””ì˜¤ íŠ¸ë™ì„ í™”ë©´ ê³µìœ ë¡œ êµì²´
        const videoTrack = screenStream.getVideoTracks()[0];

            s => s.track?.kind === 'video'
        );

        if (sender) {
            await sender.replaceTrack(videoTrack);
        }

        // í™”ë©´ ê³µìœ  ì¤‘ë‹¨ ê°ì§€
        videoTrack.onended = () => {
            console.log('Screen sharing stopped');

        };

        return screenStream;

            console.log('User denied screen sharing');
        }
        throw error;
    }
}

// DataChannelë¡œ íŒŒì¼ ì „ì†¡
function setupDataChannel() {
    const dataChannel = peerConnection.createDataChannel('fileTransfer', {
        ordered: true,  // ìˆœì„œ ë³´ì¥
        maxRetransmits: 3  // ì¬ì „ì†¡ íšŸìˆ˜
    });

    dataChannel.onopen = () => {
        console.log('DataChannel opened');
    };

    dataChannel.onmessage = (event) => {
        // ìˆ˜ì‹ ëœ ë°ì´í„° ì²˜ë¦¬
        const data = JSON.parse(event.data);
        if (data.type === 'file-chunk') {
            handleFileChunk(data);
        }
    };

    // íŒŒì¼ì„ ì²­í¬ë¡œ ë‚˜ëˆ„ì–´ ì „ì†¡
    async function sendFile(file) {
        const chunkSize = 16384;  // 16KB ì²­í¬
        const reader = new FileReader();
        let offset = 0;

        // íŒŒì¼ ë©”íƒ€ë°ì´í„° ì „ì†¡
        dataChannel.send(JSON.stringify({
            type: 'file-start',
            name: file.name,
            size: file.size,
            mimeType: file.type
        }));

        // ì²­í¬ ë‹¨ìœ„ë¡œ ì „ì†¡
        while (offset < file.size) {
            const chunk = file.slice(offset, offset + chunkSize);
            const buffer = await chunk.arrayBuffer();

            dataChannel.send(JSON.stringify({
                type: 'file-chunk',
                data: Array.from(new Uint8Array(buffer)),
                offset
            }));

            offset += chunkSize;
        }

        dataChannel.send(JSON.stringify({ type: 'file-end' }));
    }

    return { dataChannel, sendFile };
}
