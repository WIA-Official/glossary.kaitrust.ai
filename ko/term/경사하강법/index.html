import numpy as np

def gradient_descent_from_scratch():
    """ì„ í˜• íšŒê·€ë¥¼ ìœ„í•œ ê²½ì‚¬í•˜ê°•ë²• êµ¬í˜„"""

    # 1. ë°ì´í„° ìƒì„± (y = 2x + 1 + noise)

    # 2. íŒŒë¼ë¯¸í„° ì´ˆê¸°í™”
    w = np.random.randn(1, 1)  # ê°€ì¤‘ì¹˜
