from cassandra.concurrent import execute_concurrent_with_args

def get_activities_for_users(user_ids: list, target_date: date, limit: int = 100):
    """ì—¬ëŸ¬ ì‚¬ìš©ìì˜ í™œë™ì„ ë™ì‹œ ì¡°íšŒ"""
