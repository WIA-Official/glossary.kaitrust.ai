import torch
import torch.nn as nn
from transformers import AutoModel

# 1. κ°„λ‹¨ν• μ‹ κ²½λ§ νλΌλ―Έν„° μμ 
class SimpleNN(nn.Module):
    def __init__(self, input_dim=784, hidden_dim=256, output_dim=10):
        super().__init__()
        self.fc1 = nn.Linear(input_dim, hidden_dim)  # κ°€μ¤‘μΉ + νΈν–¥
        self.fc2 = nn.Linear(hidden_dim, output_dim)

    def forward(self, x):
        return self.fc2(torch.relu(self.fc1(x)))

model = SimpleNN()

# νλΌλ―Έν„° μ κ³„μ‚°
def count_parameters(model):
    total = sum(p.numel() for p in model.parameters())
    trainable = sum(p.numel() for p in model.parameters() if p.requires_grad)
    return total, trainable

total, trainable = count_parameters(model)
print(f"Total: {total:,}, Trainable: {trainable:,}")
# μ¶λ ¥: Total: 203,530, Trainable: 203,530
# fc1: 784*256 + 256 = 200,960
# fc2: 256*10 + 10 = 2,570

# 2. LLM νλΌλ―Έν„° ν™•μΈ
llm = AutoModel.from_pretrained("bert-base-uncased")

print(f"BERT-base parameters: {total_llm / 1e6:.1f}M")
# μ¶λ ¥: BERT-base parameters: 109.5M
