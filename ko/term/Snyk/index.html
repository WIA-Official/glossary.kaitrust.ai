import requests
import json
from typing import List, Dict

class SnykClient:
    """Snyk API í´ë¼ì´ì–¸íŠ¸"""

    def __init__(self, token: str, org_id: str):
        self.base_url = "https://api.snyk.io/v1"
        self.headers = {
            "Authorization": f"token {token}",
            "Content-Type": "application/json"
        }
        self.org_id = org_id

    def list_projects(self) -> List[Dict]:
        """ì¡°ì§ì˜ ëª¨ë“  í”„ë¡œì íŠ¸ ëª©ë¡ ì¡°íšŒ"""
        response = requests.get(
            f"{self.base_url}/org/{self.org_id}/projects",
            headers=self.headers
        )
        response.raise_for_status()
        return response.json()["projects"]

    def get_project_issues(self, project_id: str) -> Dict:
        """í”„ë¡œì íŠ¸ì˜ ì·¨ì•½ì  ìƒì„¸ ì¡°íšŒ"""
        response = requests.post(
            f"{self.base_url}/org/{self.org_id}/project/{project_id}/issues",
            headers=self.headers,
            json={
                "filters": {
                    "severities": ["critical", "high"],
                    "types": ["vuln"],
                    "ignored": False,
                    "patched": False
                }
            }
        )
        response.raise_for_status()
        return response.json()

    def ignore_issue(self, project_id: str, issue_id: str, reason: str):
        """íŠ¹ì • ì´ìŠˆ ë¬´ì‹œ ì²˜ë¦¬"""
        response = requests.post(
            f"{self.base_url}/org/{self.org_id}/project/{project_id}/ignore/{issue_id}",
            headers=self.headers,
            json={
                "ignorePath": "",
                "reason": reason,
                "reasonType": "not-vulnerable",
                "disregardIfFixable": False
            }
        )
        response.raise_for_status()
        return response.json()

    def generate_report(self) -> Dict:
        """ì¡°ì§ ì „ì²´ ë³´ì•ˆ ë¦¬í¬íŠ¸ ìƒì„±"""
        projects = self.list_projects()
        report = {
            "total_projects": len(projects),
            "critical_issues": 0,
            "high_issues": 0,
            "projects_with_issues": []
        }

        for project in projects:
            issues = self.get_project_issues(project["id"])
            issue_count = issues.get("issues", {})

            critical = len([i for i in issue_count.get("vulnerabilities", [])
                           if i["severity"] == "critical"])
            high = len([i for i in issue_count.get("vulnerabilities", [])
                       if i["severity"] == "high"])

            if critical > 0 or high > 0:
                report["critical_issues"] += critical
                report["high_issues"] += high
                report["projects_with_issues"].append({
                    "name": project["name"],
                    "critical": critical,
                    "high": high
                })

        return report
