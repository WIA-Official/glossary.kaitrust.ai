    defer n.mu.Unlock()

    reply.Success = false

    // ì˜¤ë˜ëœ Leaderì˜ ìš”ì²­ì€ ê±°ë¶€

           n.log[args.PrevLogIndex].Term != args.PrevLogTerm {

            }

    }

}
