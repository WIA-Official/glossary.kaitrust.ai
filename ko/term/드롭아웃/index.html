import torch.nn as nn
import torch.nn.functional as F

class TransformerBlockWithDropout(nn.Module):
    """GPT/BERT ìŠ¤íƒ€ì¼ Transformer ë¸”ë¡ì˜ Dropout ì ìš© ìœ„ì¹˜"""
    def __init__(self, d_model=768, n_heads=12, dropout=0.1):

        self.norm1 = nn.LayerNorm(d_model)
        self.norm2 = nn.LayerNorm(d_model)
        self.ffn = nn.Sequential(
            nn.Linear(d_model, d_model * 4),
            nn.GELU(),
            nn.Dropout(dropout),  # FFN ì¤‘ê°„ì— Dropout
            nn.Linear(d_model * 4, d_model),
            nn.Dropout(dropout)   # FFN ì¶œë ¥ì— Dropout
        )
        self.dropout = nn.Dropout(dropout)

    def forward(self, x, mask=None):
        # Attention í›„ Dropout + Residual
        attn_out, _ = self.attention(x, x, x, attn_mask=mask)
        x = self.norm1(x + self.dropout(attn_out))

        # FFN í›„ Residual
        ffn_out = self.ffn(x)
        x = self.norm2(x + ffn_out)

class ScheduledDropout(nn.Module):
    def __init__(self, max_dropout=0.5, warmup_steps=1000):
        super().__init__()
        self.max_dropout = max_dropout
        self.warmup_steps = warmup_steps

        self.current_step += 1
