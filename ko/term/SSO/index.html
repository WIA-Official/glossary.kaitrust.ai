async function setupOIDC() {

                scope: 'openid profile email',

            // - ì„œëª… ê²€ì¦ (JWKS ì‚¬ìš©)

            // - ëŒ€ìƒ(aud) í™•ì¸

@app.route('/saml/acs', methods=['POST'])
def saml_acs():
    """Assertion Consumer Service - IdP ì‘ë‹µ ì²˜ë¦¬"""
    req = prepare_flask_request(request)
    auth = init_saml_auth(req)

        return f'SAML Error: {", ".join(errors)}', 400

        return 'Not authenticated', 401

    # SAML Assertionì—ì„œ ì‚¬ìš©ì ì •ë³´ ì¶”ì¶œ

        'name_id': auth.get_nameid(),
        'name_id_format': auth.get_nameid_format(),

        'attributes': auth.get_attributes(),
    }

    # ì‚¬ìš©ì ì†ì„± ë§¤í•‘ (IdP ì„¤ì •ì— ë”°ë¼ ë‹¤ë¦„)
    attrs = auth.get_attributes()

        'name': attrs.get('displayName', [None])[0],
        'department': attrs.get('department', [None])[0],
        'roles': attrs.get('memberOf', []),
    }

    # ì›ë˜ ìš”ì²­ URLë¡œ ë¦¬ë‹¤ì´ë ‰íŠ¸
    relay_state = request.form.get('RelayState', url_for('dashboard'))

@app.route('/saml/slo', methods=['GET', 'POST'])
def saml_slo():

    auth = init_saml_auth(req)

    if request.method == 'GET':
        # SP ì‹œì‘ SLO

            name_id=name_id,
            session_index=session_index,
            return_to=url_for('index')
        ))
    else:
        # IdP ì‹œì‘ SLO ì‘ë‹µ ì²˜ë¦¬
        auth.process_slo()
        session.clear()
        return redirect(url_for('index'))

@Configuration
@EnableWebSecurity
public class SsoSecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .authorizeHttpRequests(auth -> auth
                // ê³µê°œ ê²½ë¡œ
                .requestMatchers("/", "/login", "/error").permitAll()
                // ê´€ë¦¬ì ì „ìš©
                .requestMatchers("/admin/**").hasRole("ADMIN")
                // ë‚˜ë¨¸ì§€ëŠ” ì¸ì¦ í•„ìš”
                .anyRequest().authenticated()
            )
            .oauth2Login(oauth2 -> oauth2
                // ì»¤ìŠ¤í…€ ë¡œê·¸ì¸ í˜ì´ì§€
                .loginPage("/login")
                // ë¡œê·¸ì¸ ì„±ê³µ í›„ ë¦¬ë‹¤ì´ë ‰íŠ¸
                .defaultSuccessUrl("/dashboard", true)
                // ì»¤ìŠ¤í…€ UserServiceë¡œ ê¶Œí•œ ë§¤í•‘
                .userInfoEndpoint(userInfo -> userInfo
                    .oidcUserService(oidcUserService())
                )
            )
            .logout(logout -> logout
                .logoutSuccessUrl("/")
                // OIDC RP-Initiated Logout
                .logoutSuccessHandler(oidcLogoutSuccessHandler())
            );

        return http.build();
    }

    @Bean
    public OAuth2UserService oidcUserService() {
        final OidcUserService delegate = new OidcUserService();

        return userRequest -> {
            // ê¸°ë³¸ OIDC ì‚¬ìš©ì ì •ë³´ ë¡œë“œ
            OidcUser oidcUser = delegate.loadUser(userRequest);

            // IdPì˜ ê·¸ë£¹/ì—­í•  í´ë ˆì„ì„ Spring Security ê¶Œí•œìœ¼ë¡œ ë§¤í•‘
            Set mappedAuthorities = new HashSet<>();

            // ê¸°ë³¸ ê¶Œí•œ
            mappedAuthorities.add(new SimpleGrantedAuthority("ROLE_USER"));

            // IdP ê·¸ë£¹ì„ ì—­í• ë¡œ ë³€í™˜ (ì˜ˆ: Okta groups í´ë ˆì„)
            List groups = oidcUser.getAttribute("groups");
            if (groups != null) {
                groups.stream()
                    .map(group -> new SimpleGrantedAuthority("ROLE_" + group.toUpperCase()))
                    .forEach(mappedAuthorities::add);
            }

            // ì»¤ìŠ¤í…€ í´ë ˆì„ ì²˜ë¦¬ (ì˜ˆ: ë¶€ì„œë³„ ê¶Œí•œ)
            String department = oidcUser.getAttribute("department");
            if ("IT".equals(department)) {
                mappedAuthorities.add(new SimpleGrantedAuthority("ROLE_ADMIN"));
            }

            return new DefaultOidcUser(
                mappedAuthorities,
                oidcUser.getIdToken(),
                oidcUser.getUserInfo()
            );
        };
    }

    @Bean
    public OidcClientInitiatedLogoutSuccessHandler oidcLogoutSuccessHandler() {
        OidcClientInitiatedLogoutSuccessHandler handler =
            new OidcClientInitiatedLogoutSuccessHandler(clientRegistrationRepository);
        handler.setPostLogoutRedirectUri("{baseUrl}");
        return handler;
    }
}
