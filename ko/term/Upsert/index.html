def upsert_user_postgres(session, email: str, name: str) -> User:
    """PostgreSQL ON CONFLICT DO UPDATE"""
    stmt = pg_insert(User).values(
        email=email,
        name=name,
        login_count=1,
        last_login=datetime.utcnow()
    )

    # ì¶©ëŒ ì‹œ ì—…ë°ì´íŠ¸í•  ì»¬ëŸ¼ ì§€ì •
    stmt = stmt.on_conflict_do_update(
        index_elements=['email'],  # ì¶©ëŒ ê°ì§€ ì»¬ëŸ¼
        set_={
            'name': stmt.excluded.name,
            'login_count': User.login_count + 1,
            'last_login': stmt.excluded.last_login,
            'updated_at': datetime.utcnow()
        }
    ).returning(User)

    result = session.execute(stmt)
    session.commit()
    return result.scalar_one()

def upsert_user_mysql(session, email: str, name: str) -> int:
    """MySQL ON DUPLICATE KEY UPDATE"""
    stmt = mysql_insert(User).values(
        email=email,
        name=name,
        login_count=1,
        last_login=datetime.utcnow()

    stmt = stmt.on_duplicate_key_update(
        name=stmt.inserted.name,
        login_count=User.login_count + 1,
        last_login=stmt.inserted.last_login,
        updated_at=datetime.utcnow()
    )

    result = session.execute(stmt)
    session.commit()
    return result.lastrowid
