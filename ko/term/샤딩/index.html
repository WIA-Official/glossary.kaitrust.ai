from typing import Dict, Any, List
import psycopg2

class ShardRouter:
    """ìƒ¤ë“œ ë¼ìš°í„° - ë°ì´í„°ë¥¼ ì ì ˆí•œ ìƒ¤ë“œë¡œ ë¼ìš°íŒ…"""

    def __init__(self, shard_configs: List[Dict[str, str]]):
        """
        Args:
            shard_configs: ìƒ¤ë“œë³„ ì—°ê²° ì •ë³´ ë¦¬ìŠ¤íŠ¸
            [{"host": "shard1.db.com", "dbname": "app_db"}, ...]
        """
        self.shards = []
        self.num_shards = len(shard_configs)

        for config in shard_configs:
            conn = psycopg2.connect(**config)
            self.shards.append(conn)

    def get_shard_id(self, shard_key: str) -> int:
        """
        í•´ì‹œ ê¸°ë°˜ ìƒ¤ë“œ ID ê²°ì •
        ì¼ê´€ëœ í•´ì‹±ìœ¼ë¡œ ê°™ì€ í‚¤ëŠ” í•­ìƒ ê°™ì€ ìƒ¤ë“œë¡œ
        """
        hash_value = int(hashlib.md5(str(shard_key).encode()).hexdigest(), 16)
        return hash_value % self.num_shards

    def get_connection(self, shard_key: str):
        """ìƒ¤ë“œ í‚¤ì— í•´ë‹¹í•˜ëŠ” DB ì»¤ë„¥ì…˜ ë°˜í™˜"""
        shard_id = self.get_shard_id(shard_key)
        return self.shards[shard_id]

    def execute_on_shard(self, shard_key: str, query: str, params: tuple = None):
        """íŠ¹ì • ìƒ¤ë“œì—ì„œ ì¿¼ë¦¬ ì‹¤í–‰"""
        conn = self.get_connection(shard_key)
        with conn.cursor() as cur:
            cur.execute(query, params)
            conn.commit()
            return cur.fetchall() if cur.description else None

    def execute_on_all_shards(self, query: str, params: tuple = None) -> List:
        """ëª¨ë“  ìƒ¤ë“œì—ì„œ ì¿¼ë¦¬ ì‹¤í–‰ (ì§‘ê³„ ë“±)"""
        results = []
        for shard in self.shards:
            with shard.cursor() as cur:
                cur.execute(query, params)
                if cur.description:
                    results.extend(cur.fetchall())
        return results

    def create_user(self, user_id: str, name: str, email: str):
        """
        ì‚¬ìš©ì ìƒì„± - user_idê°€ ìƒ¤ë“œ í‚¤
        ê°™ì€ user_idëŠ” í•­ìƒ ê°™ì€ ìƒ¤ë“œì— ì €ì¥ë¨

            INSERT INTO users (user_id, name, email, created_at)

    def get_user(self, user_id: str) -> Dict[str, Any]:
        """ì‚¬ìš©ì ì¡°íšŒ - ìƒ¤ë“œ í‚¤ë¡œ ì •í™•í•œ ìƒ¤ë“œ ì ‘ê·¼"""

        result = self.router.execute_on_shard(user_id, query, (user_id,))
        return result[0] if result else None

    def get_user_count(self) -> int:
        """ì „ì²´ ì‚¬ìš©ì ìˆ˜ - ëª¨ë“  ìƒ¤ë“œì—ì„œ ì§‘ê³„"""

class ConsistentHashRing:
    """
    ì¼ê´€ëœ í•´ì‹± - ìƒ¤ë“œ ì¶”ê°€/ì œê±° ì‹œ ìµœì†Œí•œì˜ ë°ì´í„° ì´ë™
    """

    def __init__(self, nodes: List[str], virtual_nodes: int = 150):
        self.ring = {}
        self.sorted_keys = []
        self.virtual_nodes = virtual_nodes

        for node in nodes:
            self.add_node(node)

    def _hash(self, key: str) -> int:
        return int(hashlib.md5(key.encode()).hexdigest(), 16)

    def add_node(self, node: str):
        """ë…¸ë“œ ì¶”ê°€ - ê°€ìƒ ë…¸ë“œë¡œ ê· ë“± ë¶„ì‚°"""
        for i in range(self.virtual_nodes):
            virtual_key = f"{node}:{i}"
            hash_key = self._hash(virtual_key)
            self.ring[hash_key] = node
            self.sorted_keys.append(hash_key)
        self.sorted_keys.sort()

    def remove_node(self, node: str):
        """ë…¸ë“œ ì œê±°"""
        for i in range(self.virtual_nodes):
            virtual_key = f"{node}:{i}"
            hash_key = self._hash(virtual_key)
            del self.ring[hash_key]
            self.sorted_keys.remove(hash_key)

    def get_node(self, key: str) -> str:
        """í‚¤ì— í•´ë‹¹í•˜ëŠ” ë…¸ë“œ ì°¾ê¸°"""
        if not self.ring:
            return None

        hash_key = self._hash(key)

        # ì´ì§„ íƒìƒ‰ìœ¼ë¡œ ì ì ˆí•œ ë…¸ë“œ ì°¾ê¸°
        for ring_key in self.sorted_keys:
            if hash_key <= ring_key:
                return self.ring[ring_key]

        # ë§ì˜ ì²˜ìŒìœ¼ë¡œ ìˆœí™˜
        return self.ring[self.sorted_keys[0]]
