    """Elastic Weight Consolidation - Continual Learning ê¸°ë²•"""

    def __init__(self, model: nn.Module, dataloader, device, lambda_ewc: float = 1000):
        self.model = model
        self.lambda_ewc = lambda_ewc
        self.device = device

        # ì´ì „ íƒœìŠ¤í¬ í•™ìŠµ í›„ íŒŒë¼ë¯¸í„° ì €ì¥
        self.params_old = {n: p.clone().detach()

        # Fisher Information Matrix ê³„ì‚° (íŒŒë¼ë¯¸í„° ì¤‘ìš”ë„)
        self.fisher = self._compute_fisher(dataloader)

    def _compute_fisher(self, dataloader) -> dict:
        """Fisher Information ê·¼ì‚¬ ê³„ì‚°"""
        fisher = {n: torch.zeros_like(p)
                  for n, p in self.model.named_parameters() if p.requires_grad}

        self.model.eval()
        for data, target in dataloader:
            data, target = data.to(self.device), target.to(self.device)
            self.model.zero_grad()
            output = self.model(data)
            loss = nn.functional.cross_entropy(output, target)
            loss.backward()

            for n, p in self.model.named_parameters():
                if p.requires_grad and p.grad is not None:
                    fisher[n] += p.grad.data ** 2

        # í‰ê· í™”
        for n in fisher:
            fisher[n] /= len(dataloader)

        return fisher
