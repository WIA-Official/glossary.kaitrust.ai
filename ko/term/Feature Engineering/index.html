                                     '2024-01-10', '2024-01-20', '2024-02-15']),
    'amount': [100, 250, 180, 500, 120, 80],
    'category': ['electronics', 'clothing', 'electronics', 'food', 'food', 'clothing'],
    'is_weekend': [False, False, True, True, False, True]
