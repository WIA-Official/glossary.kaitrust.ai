CREATE VIEW order_summary AS
SELECT
    o.id AS order_id,
    c.name AS customer_name,
    SUM(oi.quantity * oi.price) AS total,
    o.created_at
FROM orders o
JOIN customers c ON o.customer_id = c.id
JOIN order_items oi ON o.id = oi.order_id
GROUP BY o.id, c.name, o.created_at;
