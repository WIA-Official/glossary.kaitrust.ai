import torch
import torch.nn as nn

class DeepNeuralNetwork(nn.Module):

            nn.Linear(28*28, 512),   # ì…ë ¥ì¸µ â†’ ì€ë‹‰ì¸µ1
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(512, 256),      # ì€ë‹‰ì¸µ1 â†’ ì€ë‹‰ì¸µ2

            nn.Linear(256, 128),      # ì€ë‹‰ì¸µ2 â†’ ì€ë‹‰ì¸µ3

# ë°ì´í„° ë¡œë“œ (MNIST)

    transforms.ToTensor(),
    transforms.Normalize((0.1307,), (0.3081,))

    total_loss = 0

        optimizer.zero_grad()
