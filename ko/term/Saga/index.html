class SagaStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    COMPENSATING = "compensating"
    FAILED = "failed"


@dataclass
class SagaStep:
    """Sagaì˜ ê°œë³„ ë‹¨ê³„"""
    name: str
    action: Callable  # ì‹¤í–‰ í•¨ìˆ˜
    compensation: Callable  # ë³´ìƒ í•¨ìˆ˜

class SagaOrchestrator:
    """Saga Orchestrator - ì „ì²´ íë¦„ ì œì–´"""

    def __init__(self, saga_id: str = None):
        self.saga_id = saga_id or str(uuid.uuid4())
        self.steps: List[SagaStep] = []
        self.completed_steps: List[SagaStep] = []
        self.status = SagaStatus.PENDING
        self.context = {}  # ë‹¨ê³„ ê°„ ë°ì´í„° ê³µìœ 

    def add_step(self, name: str, action: Callable, compensation: Callable):
        """Saga ë‹¨ê³„ ì¶”ê°€"""
        self.steps.append(SagaStep(name, action, compensation))
        return self

    def execute(self, initial_context: dict = None) -> dict:
        """Saga ì‹¤í–‰"""
        self.context = initial_context or {}
        self.status = SagaStatus.RUNNING
        logger.info(f"ğŸš€ Saga ì‹œì‘: {self.saga_id}")

        try:
            for step in self.steps:
                logger.info(f"  â–¶ ë‹¨ê³„ ì‹¤í–‰: {step.name}")
                step.status = SagaStatus.RUNNING

                # ë©±ë“±ì„±ì„ ìœ„í•œ íŠ¸ëœì­ì…˜ ID
                tx_id = f"{self.saga_id}-{step.name}"
                self.context['current_tx_id'] = tx_id

                # ë‹¨ê³„ ì‹¤í–‰
                result = step.action(self.context)
                self.context.update(result or {})

                step.status = SagaStatus.COMPLETED
                self.completed_steps.append(step)
                logger.info(f"  âœ… ë‹¨ê³„ ì™„ë£Œ: {step.name}")

            self.status = SagaStatus.COMPLETED
            logger.info(f"ğŸ‰ Saga ì™„ë£Œ: {self.saga_id}")
            return {"status": "success", "saga_id": self.saga_id, "context": self.context}

        except Exception as e:
            logger.error(f"  âŒ ë‹¨ê³„ ì‹¤íŒ¨: {e}")
            self._compensate()
            return {"status": "failed", "saga_id": self.saga_id, "error": str(e)}

    def _compensate(self):
        """ë³´ìƒ íŠ¸ëœì­ì…˜ ì‹¤í–‰ (ì—­ìˆœ)"""
        self.status = SagaStatus.COMPENSATING
        logger.info(f"ğŸ”„ ë³´ìƒ íŠ¸ëœì­ì…˜ ì‹œì‘ (ì™„ë£Œëœ {len(self.completed_steps)}ê°œ ë‹¨ê³„)")

        for step in reversed(self.completed_steps):
            try:
                logger.info(f"  â†© ë³´ìƒ ì‹¤í–‰: {step.name}")
                step.compensation(self.context)
                logger.info(f"  âœ… ë³´ìƒ ì™„ë£Œ: {step.name}")
            except Exception as e:
                # ë³´ìƒ ì‹¤íŒ¨ ì‹œ ìˆ˜ë™ ê°œì… í•„ìš”
                logger.error(f"  âš ï¸ ë³´ìƒ ì‹¤íŒ¨: {step.name} - {e}")
                self._save_for_manual_intervention(step, e)

        self.status = SagaStatus.FAILED
        logger.info(f"ğŸ”š Saga ë¡¤ë°± ì™„ë£Œ: {self.saga_id}")

    def _save_for_manual_intervention(self, step: SagaStep, error: Exception):
        """ìˆ˜ë™ ê°œì…ì´ í•„ìš”í•œ ì¼€ì´ìŠ¤ ì €ì¥"""
        logger.warning(f"ğŸ“‹ ìˆ˜ë™ ì²˜ë¦¬ í•„ìš”: Saga={self.saga_id}, Step={step.name}")

            workflow.logger.info("ì¬ê³  ì˜ˆì•½ ì™„ë£Œ")

            # 3. ê²°ì œ ì²˜ë¦¬
            data.payment_id = await workflow.execute_activity(
                process_payment,
                data,
                start_to_close_timeout=timedelta(seconds=60),
                retry_policy=retry_policy

            workflow.logger.info(f"ê²°ì œ ì™„ë£Œ: {data.payment_id}")
