import mlflow
import mlflow.pytorch
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset

# MLflow ì„¤ì •
mlflow.set_tracking_uri("sqlite:///mlflow.db")
mlflow.set_experiment("pytorch_classifier")

# ê°„ë‹¨í•œ ëª¨ë¸ ì •ì˜
class SimpleNet(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim):
        super().__init__()
        self.fc1 = nn.Linear(input_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, output_dim)
        self.relu = nn.ReLU()

    def forward(self, x):
        return self.fc2(self.relu(self.fc1(x)))

# í•˜ì´í¼íŒŒë¼ë¯¸í„°
config = {
    "input_dim": 20,
    "hidden_dim": 64,
    "output_dim": 2,
    "learning_rate": 0.001,
    "epochs": 10,
    "batch_size": 32
}

# MLflow autolog í™œì„±í™” (PyTorch ìë™ ì¶”ì )
mlflow.pytorch.autolog()

with mlflow.start_run(run_name="pytorch_simple_net"):
    # í•˜ì´í¼íŒŒë¼ë¯¸í„° ë¡œê¹…
    mlflow.log_params(config)

    # ëª¨ë¸ ì´ˆê¸°í™”
    model = SimpleNet(config["input_dim"], config["hidden_dim"], config["output_dim"])
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=config["learning_rate"])

    # ë”ë¯¸ ë°ì´í„° ìƒì„±
    X = torch.randn(1000, config["input_dim"])
    y = torch.randint(0, 2, (1000,))
    dataset = TensorDataset(X, y)
    loader = DataLoader(dataset, batch_size=config["batch_size"])

    # í•™ìŠµ ë£¨í”„
    for epoch in range(config["epochs"]):
        total_loss = 0
        for batch_X, batch_y in loader:
            optimizer.zero_grad()
            outputs = model(batch_X)
            loss = criterion(outputs, batch_y)
            loss.backward()
            optimizer.step()
            total_loss += loss.item()

        avg_loss = total_loss / len(loader)
        mlflow.log_metric("train_loss", avg_loss, step=epoch)
        print(f"Epoch {epoch+1}: Loss = {avg_loss:.4f}")

    # ëª¨ë¸ ì €ì¥
    mlflow.pytorch.log_model(model, "model")

    # ëª¨ë¸ ë ˆì§€ìŠ¤íŠ¸ë¦¬ì— ë“±ë¡
    run_id = mlflow.active_run().info.run_id
    model_uri = f"runs:/{run_id}/model"
    mlflow.register_model(model_uri, "SimpleClassifier")

print("MLflow UI: http://localhost:5000")
