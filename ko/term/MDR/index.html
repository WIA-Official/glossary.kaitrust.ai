    def _sign_request(self, method: str, path: str, body: str = "") -> Dict[str, str]:
        """API ìš”ì²­ ì„œëª… ìƒì„±"""

            self.api_secret.encode(),
            string_to_sign.encode(),
            hashlib.sha256

            "X-API-Key": self.api_key,

            f"{self.base_url}{path}",
