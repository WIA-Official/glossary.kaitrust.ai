# t-SNE ì™„ì „ ê°€ì´ë“œ: ì‹œê°í™”, ìµœì í™”, ë¹„êµ
import numpy as np
import matplotlib.pyplot as plt
from sklearn.manifold import TSNE
from sklearn.datasets import load_digits, fetch_20newsgroups
from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
import time

# 1. ê¸°ë³¸ t-SNE ì‹œê°í™”
def basic_tsne_visualization():
    """MNIST ìˆ«ì ë°ì´í„° t-SNE ì‹œê°í™”"""
    # ë°ì´í„° ë¡œë“œ
    digits = load_digits()
    X, y = digits.data, digits.target

    print(f"ë°ì´í„° í˜•íƒœ: {X.shape}")  # (1797, 64)

    # t-SNE ì ìš©
    tsne = TSNE(
        n_components=2,      # 2Dë¡œ ì¶•ì†Œ
        perplexity=30,       # ìœ íš¨ ì´ì›ƒ ìˆ˜
        learning_rate=200,   # í•™ìŠµë¥ 
        n_iter=1000,         # ë°˜ë³µ íšŸìˆ˜
        random_state=42
    )

    start_time = time.time()
    X_tsne = tsne.fit_transform(X)
    elapsed = time.time() - start_time
    print(f"t-SNE ì‹œê°„: {elapsed:.2f}ì´ˆ")

    # ì‹œê°í™”
    plt.figure(figsize=(12, 10))
    scatter = plt.scatter(
        X_tsne[:, 0], X_tsne[:, 1],
        c=y,
        cmap='tab10',
        alpha=0.7,
        s=20
    )
    plt.colorbar(scatter, label='ìˆ«ì í´ë˜ìŠ¤')
    plt.title('MNIST Digits t-SNE Visualization')
    plt.xlabel('t-SNE 1')
    plt.ylabel('t-SNE 2')
    plt.savefig('tsne_mnist.png', dpi=150)
    plt.close()

    return X_tsne

# 2. Perplexity ê°’ì— ë”°ë¥¸ ê²°ê³¼ ë¹„êµ
def compare_perplexity():
    """ë‹¤ì–‘í•œ perplexityë¡œ t-SNE ê²°ê³¼ ë¹„êµ"""
    digits = load_digits()
    X, y = digits.data, digits.target

    perplexities = [5, 15, 30, 50, 100]

    fig, axes = plt.subplots(1, 5, figsize=(25, 5))

    for ax, perp in zip(axes, perplexities):
        tsne = TSNE(
            n_components=2,
            perplexity=perp,
            random_state=42
        )
        X_tsne = tsne.fit_transform(X)

        ax.scatter(X_tsne[:, 0], X_tsne[:, 1], c=y, cmap='tab10', s=10, alpha=0.7)
        ax.set_title(f'Perplexity = {perp}')
        ax.set_xticks([])
        ax.set_yticks([])

    plt.tight_layout()
    plt.savefig('tsne_perplexity_comparison.png', dpi=150)
    plt.close()

# 3. ëŒ€ìš©ëŸ‰ ë°ì´í„°: PCA ì „ì²˜ë¦¬ + t-SNE
def optimized_tsne_pipeline():
    """ëŒ€ìš©ëŸ‰ ë°ì´í„°ë¥¼ ìœ„í•œ PCA + t-SNE íŒŒì´í”„ë¼ì¸"""
    # ê°€ìƒì˜ ê³ ì°¨ì› ë°ì´í„° (10000 ìƒ˜í”Œ, 784 ì°¨ì›)
    np.random.seed(42)
    n_samples = 10000
    n_features = 784

    # 3ê°œ í´ëŸ¬ìŠ¤í„° ìƒì„±
    X1 = np.random.randn(n_samples // 3, n_features) + np.array([5] * n_features)
    X2 = np.random.randn(n_samples // 3, n_features) + np.array([-5] * n_features)
    X3 = np.random.randn(n_samples // 3, n_features)
    X = np.vstack([X1, X2, X3])
    y = np.array([0] * (n_samples // 3) + [1] * (n_samples // 3) + [2] * (n_samples // 3))

    print(f"ì›ë³¸ ë°ì´í„°: {X.shape}")

    # Step 1: ìŠ¤ì¼€ì¼ë§
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)

    # Step 2: PCAë¡œ ë¨¼ì € ì°¨ì› ì¶•ì†Œ (t-SNE ì†ë„ í–¥ìƒ)
    pca = PCA(n_components=50)
    X_pca = pca.fit_transform(X_scaled)
    print(f"PCA í›„: {X_pca.shape}, ì„¤ëª… ë¶„ì‚°: {pca.explained_variance_ratio_.sum():.2%}")

    # Step 3: t-SNE ì ìš©
    start = time.time()
    tsne = TSNE(
        n_components=2,
        perplexity=30,
        n_iter=500,
        init='pca',  # PCA ì´ˆê¸°í™” (ë” ì•ˆì •ì )
        random_state=42
    )
    X_tsne = tsne.fit_transform(X_pca)
    print(f"t-SNE ì‹œê°„: {time.time() - start:.2f}ì´ˆ")

    # ì‹œê°í™”
    plt.figure(figsize=(10, 8))
    plt.scatter(X_tsne[:, 0], X_tsne[:, 1], c=y, cmap='tab10', s=5, alpha=0.5)
    plt.title('PCA + t-SNE Pipeline')
    plt.savefig('tsne_optimized.png', dpi=150)
    plt.close()

    return X_tsne

# 4. t-SNE vs UMAP ë¹„êµ
def tsne_vs_umap():
    """t-SNEì™€ UMAP ì„±ëŠ¥ ë° ê²°ê³¼ ë¹„êµ"""
    try:
        import umap

        digits = load_digits()
        X, y = digits.data, digits.target

        # t-SNE
        start = time.time()
        tsne = TSNE(n_components=2, random_state=42)
        X_tsne = tsne.fit_transform(X)
        tsne_time = time.time() - start

        # UMAP
        start = time.time()
        umap_model = umap.UMAP(n_components=2, random_state=42)
        X_umap = umap_model.fit_transform(X)
        umap_time = time.time() - start

        print(f"t-SNE ì‹œê°„: {tsne_time:.2f}ì´ˆ")
        print(f"UMAP ì‹œê°„: {umap_time:.2f}ì´ˆ")
        print(f"UMAPì´ {tsne_time/umap_time:.1f}ë°° ë¹ ë¦„")

        # ì‹œê°í™”
        fig, axes = plt.subplots(1, 2, figsize=(16, 6))

        axes[0].scatter(X_tsne[:, 0], X_tsne[:, 1], c=y, cmap='tab10', s=10, alpha=0.7)
        axes[0].set_title(f't-SNE ({tsne_time:.1f}s)')

        axes[1].scatter(X_umap[:, 0], X_umap[:, 1], c=y, cmap='tab10', s=10, alpha=0.7)
        axes[1].set_title(f'UMAP ({umap_time:.1f}s)')

        plt.tight_layout()
        plt.savefig('tsne_vs_umap.png', dpi=150)
        plt.close()

    except ImportError:
        print("UMAPì´ ì„¤ì¹˜ë˜ì§€ ì•Šì•˜ìŠµë‹ˆë‹¤: pip install umap-learn")
