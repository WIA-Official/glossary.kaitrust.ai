# k-NN ì™„ì „ ê°€ì´ë“œ: ë¶„ë¥˜, íšŒê·€, ìµœì í™”
import numpy as np
from sklearn.neighbors import KNeighborsClassifier, KNeighborsRegressor
from sklearn.model_selection import cross_val_score, GridSearchCV
from sklearn.preprocessing import StandardScaler
from sklearn.datasets import load_iris, load_diabetes
from sklearn.pipeline import Pipeline
import matplotlib.pyplot as plt

# 1. ê¸°ë³¸ k-NN ë¶„ë¥˜
def basic_knn_classification():
    """ê¸°ë³¸ k-NN ë¶„ë¥˜ ì˜ˆì œ"""
    # ë°ì´í„° ë¡œë“œ
    iris = load_iris()
    X, y = iris.data, iris.target

    # ìŠ¤ì¼€ì¼ë§ + k-NN íŒŒì´í”„ë¼ì¸
    pipeline = Pipeline([
        ('scaler', StandardScaler()),  # k-NNì€ ìŠ¤ì¼€ì¼ì— ë¯¼ê°!
        ('knn', KNeighborsClassifier(n_neighbors=5))
    ])

    # êµì°¨ ê²€ì¦ìœ¼ë¡œ ì„±ëŠ¥ í‰ê°€
    scores = cross_val_score(pipeline, X, y, cv=5)
    print(f"í‰ê·  ì •í™•ë„: {scores.mean():.3f} (+/- {scores.std()*2:.3f})")

    # í•™ìŠµ ë° ì˜ˆì¸¡
    pipeline.fit(X, y)
    new_sample = [[5.1, 3.5, 1.4, 0.2]]
    prediction = pipeline.predict(new_sample)
    probabilities = pipeline.predict_proba(new_sample)

    print(f"ì˜ˆì¸¡ í´ë˜ìŠ¤: {iris.target_names[prediction[0]]}")
    print(f"í´ë˜ìŠ¤ë³„ í™•ë¥ : {dict(zip(iris.target_names, probabilities[0]))}")

    return pipeline

# 2. ìµœì  k ê°’ ì°¾ê¸°
def find_optimal_k():
    """êµì°¨ ê²€ì¦ìœ¼ë¡œ ìµœì ì˜ k ì°¾ê¸°"""
    iris = load_iris()
    X, y = iris.data, iris.target

    k_range = range(1, 31)
    scores = []

    for k in k_range:
        knn = KNeighborsClassifier(n_neighbors=k)
        score = cross_val_score(
            knn, X, y,
            cv=10,
            scoring='accuracy'
        ).mean()
        scores.append(score)

    optimal_k = k_range[np.argmax(scores)]
    print(f"ìµœì  k: {optimal_k}, ì •í™•ë„: {max(scores):.3f}")

    # ì‹œê°í™”
    plt.figure(figsize=(10, 6))
    plt.plot(k_range, scores, 'bo-')
    plt.axvline(x=optimal_k, color='r', linestyle='--', label=f'ìµœì  k={optimal_k}')
    plt.xlabel('k (ì´ì›ƒ ìˆ˜)')
    plt.ylabel('êµì°¨ ê²€ì¦ ì •í™•ë„')
    plt.title('k ê°’ì— ë”°ë¥¸ ì •í™•ë„ ë³€í™”')
    plt.legend()
    plt.grid(True)
    plt.savefig('knn_k_selection.png')
    plt.close()

    return optimal_k

# 3. í•˜ì´í¼íŒŒë¼ë¯¸í„° ê·¸ë¦¬ë“œ ì„œì¹˜
def optimize_knn_hyperparameters():
    """GridSearchCVë¡œ í•˜ì´í¼íŒŒë¼ë¯¸í„° ìµœì í™”"""
    iris = load_iris()
    X, y = iris.data, iris.target

    pipeline = Pipeline([
        ('scaler', StandardScaler()),
        ('knn', KNeighborsClassifier())
    ])

    param_grid = {
        'knn__n_neighbors': [3, 5, 7, 9, 11],
        'knn__weights': ['uniform', 'distance'],  # ê±°ë¦¬ ê°€ì¤‘ì¹˜
        'knn__metric': ['euclidean', 'manhattan', 'minkowski']
    }

    grid_search = GridSearchCV(
        pipeline,
        param_grid,
        cv=5,
        scoring='accuracy',
        n_jobs=-1
    )

    grid_search.fit(X, y)

    print(f"ìµœì  íŒŒë¼ë¯¸í„°: {grid_search.best_params_}")
    print(f"ìµœê³  ì •í™•ë„: {grid_search.best_score_:.3f}")

    return grid_search.best_estimator_

# 4. k-NN íšŒê·€
def knn_regression():
    """k-NN íšŒê·€ ì˜ˆì œ"""
    diabetes = load_diabetes()
    X, y = diabetes.data, diabetes.target

    pipeline = Pipeline([
        ('scaler', StandardScaler()),
        ('knn', KNeighborsRegressor(n_neighbors=5, weights='distance'))
    ])

    scores = cross_val_score(
        pipeline, X, y,
        cv=5,
        scoring='neg_mean_squared_error'
    )

    rmse = np.sqrt(-scores.mean())
    print(f"RMSE: {rmse:.2f}")

    return pipeline.fit(X, y)

def knn_for_vector_search():
    """RAG ì‹œìŠ¤í…œì—ì„œ ìœ ì‚¬ ë¬¸ì„œ ê²€ìƒ‰"""
    from sklearn.neighbors import NearestNeighbors

    # ê°€ìƒì˜ ë¬¸ì„œ ì„ë² ë”© (ì‹¤ì œë¡œëŠ” OpenAI/Sentence-Transformers ì‚¬ìš©)
    documents = [

        "ìë°”ìŠ¤í¬ë¦½íŠ¸ëŠ” ì›¹ ê°œë°œì— í•„ìˆ˜ì ì…ë‹ˆë‹¤.",

        "ë”¥ëŸ¬ë‹ì€ ì‹ ê²½ë§ì„ í™œìš©í•©ë‹ˆë‹¤.",
        "k-NNì€ ë¶„ë¥˜ ì•Œê³ ë¦¬ì¦˜ì…ë‹ˆë‹¤."
    ]

    # ê°„ë‹¨í•œ TF-IDF ì„ë² ë”© (ì‹¤ì œë¡œëŠ” dense embedding ì‚¬ìš©)
    from sklearn.feature_extraction.text import TfidfVectorizer

    # k-NN ëª¨ë¸ (ì½”ì‚¬ì¸ ìœ ì‚¬ë„)
    nn_model = NearestNeighbors(
        n_neighbors=3,
        metric='cosine',
        algorithm='brute'  # ì‘ì€ ë°ì´í„°ì…‹ì—ì„œëŠ” brute force
    )
    nn_model.fit(embeddings)

    # ì¿¼ë¦¬ ê²€ìƒ‰
    query = "AIì™€ ê¸°ê³„í•™ìŠµì— ëŒ€í•´ ì•Œë ¤ì£¼ì„¸ìš”"
    query_embedding = vectorizer.transform([query]).toarray()

    distances, indices = nn_model.kneighbors(query_embedding)

    print("ìœ ì‚¬ ë¬¸ì„œ ê²€ìƒ‰ ê²°ê³¼:")
    for i, (dist, idx) in enumerate(zip(distances[0], indices[0])):
        similarity = 1 - dist  # ì½”ì‚¬ì¸ ê±°ë¦¬ë¥¼ ìœ ì‚¬ë„ë¡œ ë³€í™˜
        print(f"  {i+1}. [{similarity:.3f}] {documents[idx]}")

def approximate_nn():
    """ëŒ€ìš©ëŸ‰ ë°ì´í„°ë¥¼ ìœ„í•œ ê·¼ì‚¬ ìµœê·¼ì ‘ ì´ì›ƒ (FAISS)"""
    try:
        import faiss

        # 100ë§Œ ê°œì˜ 128ì°¨ì› ë²¡í„° (ì˜ˆì‹œ)
        d = 128  # ì°¨ì›
        nb = 100000  # ë°ì´í„°ë² ì´ìŠ¤ í¬ê¸°
        nq = 10  # ì¿¼ë¦¬ ìˆ˜

        np.random.seed(1234)
        xb = np.random.random((nb, d)).astype('float32')

        # IVF ì¸ë±ìŠ¤ (í´ëŸ¬ìŠ¤í„° ê¸°ë°˜ ê·¼ì‚¬ ê²€ìƒ‰)

        index.add(xb)    # ë²¡í„° ì¶”ê°€

        # ê²€ìƒ‰ (k=4 ì´ì›ƒ)
        k = 4
        index.nprobe = 10  # ê²€ìƒ‰í•  í´ëŸ¬ìŠ¤í„° ìˆ˜
        distances, indices = index.search(xq, k)

        print(f"ì²« ë²ˆì§¸ ì¿¼ë¦¬ì˜ 4ê°œ ì´ì›ƒ ì¸ë±ìŠ¤: {indices[0]}")
        print(f"ê±°ë¦¬: {distances[0]}")

        print("FAISSê°€ ì„¤ì¹˜ë˜ì§€ ì•Šì•˜ìŠµë‹ˆë‹¤: pip install faiss-cpu")
