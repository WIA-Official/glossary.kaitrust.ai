# 1. Pinecone í•˜ì´ë¸Œë¦¬ë“œ ê²€ìƒ‰ ì„¤ì •
def setup_hybrid_search():
    """Pinecone í•˜ì´ë¸Œë¦¬ë“œ ê²€ìƒ‰ ì¸ë±ìŠ¤ ìƒì„±"""
    pc = Pinecone()

    # í•˜ì´ë¸Œë¦¬ë“œ ê²€ìƒ‰ìš© ì¸ë±ìŠ¤ (sparse + dense ì§€ì›)
    index_name = "hybrid-search-rag"

    if index_name not in pc.list_indexes().names():

# 2. ë¬¸ì„œ ì¸ë±ì‹± (Dense + Sparse ë²¡í„°)
def index_documents(documents: list[str], index):
    """ë¬¸ì„œë¥¼ ë²¡í„°í™”í•˜ê³  ì¸ë±ì‹±"""
    # Dense ì„ë² ë”© (ì‹œë§¨í‹±)
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")

    # Sparse ì„ë² ë”© (í‚¤ì›Œë“œ - BM25)
    bm25_encoder = BM25Encoder()
    bm25_encoder.fit(documents)  # ì½”í¼ìŠ¤ì— ë§ê²Œ í•™ìŠµ
    bm25_encoder.dump("bm25_params.json")  # ì €ì¥

    # í…ìŠ¤íŠ¸ ë¶„í• 
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=500,
        chunk_overlap=50
    )

    vectors = []
    for i, doc in enumerate(documents):
        chunks = text_splitter.split_text(doc)
        for j, chunk in enumerate(chunks):
            # Dense ë²¡í„°
            dense_vec = embeddings.embed_query(chunk)
            # Sparse ë²¡í„° (BM25)
            sparse_vec = bm25_encoder.encode_documents([chunk])[0]

            vectors.append({
                "id": f"doc_{i}_chunk_{j}",
                "values": dense_vec,
                "sparse_values": sparse_vec,
                "metadata": {"text": chunk, "doc_id": i}
            })

    # ë°°ì¹˜ ì—…ì„œíŠ¸
    index.upsert(vectors=vectors, namespace="documents")
    return bm25_encoder

# 3. í•˜ì´ë¸Œë¦¬ë“œ ê²€ìƒ‰ ìˆ˜í–‰
def hybrid_search(
    query: str,
    index,
    bm25_encoder: BM25Encoder,
    top_k: int = 5,
    alpha: float = 0.7  # dense ê°€ì¤‘ì¹˜ (1-alpha = sparse)
):
    """í•˜ì´ë¸Œë¦¬ë“œ ê²€ìƒ‰ ì‹¤í–‰"""
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")

    # Dense ì¿¼ë¦¬ ë²¡í„°
    dense_query = embeddings.embed_query(query)
    # Sparse ì¿¼ë¦¬ ë²¡í„°
    sparse_query = bm25_encoder.encode_queries([query])[0]

    # í•˜ì´ë¸Œë¦¬ë“œ ê²€ìƒ‰
    results = index.query(
        vector=dense_query,
        sparse_vector=sparse_query,
        top_k=top_k,
        include_metadata=True,
        namespace="documents"
    )

    return [(match.metadata["text"], match.score) for match in results.matches]

# 4. RRF (Reciprocal Rank Fusion) ì§ì ‘ êµ¬í˜„
def reciprocal_rank_fusion(
    vector_results: list,
    keyword_results: list,
    k: int = 60  # RRF ìƒìˆ˜
) -> list:
    """ë‘ ê²€ìƒ‰ ê²°ê³¼ë¥¼ RRFë¡œ ê²°í•©"""
    fused_scores = {}

    # ë²¡í„° ê²€ìƒ‰ ê²°ê³¼ ìŠ¤ì½”ì–´ë§
    for rank, (doc_id, _) in enumerate(vector_results):
        fused_scores[doc_id] = fused_scores.get(doc_id, 0) + 1 / (k + rank + 1)

    # í‚¤ì›Œë“œ ê²€ìƒ‰ ê²°ê³¼ ìŠ¤ì½”ì–´ë§
    for rank, (doc_id, _) in enumerate(keyword_results):
        fused_scores[doc_id] = fused_scores.get(doc_id, 0) + 1 / (k + rank + 1)

    # ì ìˆ˜ ê¸°ì¤€ ì •ë ¬
    sorted_results = sorted(fused_scores.items(), key=lambda x: x[1], reverse=True)
    return sorted_results
