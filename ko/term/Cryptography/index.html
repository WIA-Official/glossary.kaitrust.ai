class AESEncryptor:
    """AES-256-GCM μ•”νΈν™” ν΄λμ¤"""

    def __init__(self, password: str):
        # λΉ„λ°€λ²νΈμ—μ„ 256λΉ„νΈ ν‚¤ νμƒ (PBKDF2)
        salt = os.urandom(16)
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,  # 256λΉ„νΈ = 32λ°”μ΄νΈ
            salt=salt,
            iterations=600000,  # 2025 OWASP κ¶μ¥
        )
        self.key = kdf.derive(password.encode())
        self.salt = salt

    def encrypt(self, plaintext: str) -> dict:
        """
        AES-256-GCM μ•”νΈν™”
        - GCM λ¨λ“: μ•”νΈν™” + μΈμ¦(λ¬΄κ²°μ„± λ³΄μ¥)μ„ λ™μ‹μ—
        - Nonce(IV): κ° μ•”νΈν™”λ§λ‹¤ κ³ μ ν•΄μ•Ό ν•¨ (μ λ€ μ¬μ‚¬μ© κΈμ§€!)
        """
        # 12λ°”μ΄νΈ nonce (GCM κ¶μ¥ ν¬κΈ°)
        nonce = os.urandom(12)

        aesgcm = AESGCM(self.key)
        ciphertext = aesgcm.encrypt(
            nonce,
            plaintext.encode('utf-8'),
            associated_data=None  # AAD: μ¶”κ°€ μΈμ¦ λ°μ΄ν„° (μµμ…)
        )

        return {
            'ciphertext': base64.b64encode(ciphertext).decode(),
            'nonce': base64.b64encode(nonce).decode(),
            'salt': base64.b64encode(self.salt).decode()
        }

    def decrypt(self, encrypted_data: dict) -> str:
        """AES-256-GCM λ³µνΈν™”"""
        nonce = base64.b64decode(encrypted_data['nonce'])
        ciphertext = base64.b64decode(encrypted_data['ciphertext'])

        aesgcm = AESGCM(self.key)
        plaintext = aesgcm.decrypt(nonce, ciphertext, None)

        return plaintext.decode('utf-8')

from cryptography.hazmat.primitives.asymmetric import rsa, padding
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
import os
import base64

class HybridEncryptor:
    """
    ν•μ΄λΈλ¦¬λ“ μ•”νΈν™” (RSA + AES)
    - RSAλ΅ λ€μΉ­ν‚¤ μ•”νΈν™” (ν‚¤ κµν™)
    - AESλ΅ μ‹¤μ  λ°μ΄ν„° μ•”νΈν™” (μ„±λ¥)
    """

    def __init__(self):
        # RSA 4096λΉ„νΈ ν‚¤ μ μƒμ„± (2048μ€ μµμ†, 4096 κ¶μ¥)
        self.private_key = rsa.generate_private_key(
            public_exponent=65537,
            key_size=4096
        )
        self.public_key = self.private_key.public_key()

    def get_public_key_pem(self) -> bytes:
        """κ³µκ°ν‚¤λ¥Ό PEM ν•μ‹μΌλ΅ λ‚΄λ³΄λ‚΄κΈ°"""
        return self.public_key.public_bytes(
            encoding=serialization.Encoding.PEM,
            format=serialization.PublicFormat.SubjectPublicKeyInfo
        )

    def encrypt(self, plaintext: str, recipient_public_key) -> dict:
        """
        ν•μ΄λΈλ¦¬λ“ μ•”νΈν™”:
        1. λλ¤ AES μ„Έμ… ν‚¤ μƒμ„±
        2. RSA-OAEPλ΅ μ„Έμ… ν‚¤ μ•”νΈν™”
        3. AES-GCMμΌλ΅ λ°μ΄ν„° μ•”νΈν™”
        """
        # 1. λλ¤ 256λΉ„νΈ AES μ„Έμ… ν‚¤ μƒμ„±
        session_key = os.urandom(32)
        nonce = os.urandom(12)

        # 2. RSA-OAEPλ΅ μ„Έμ… ν‚¤ μ•”νΈν™” (PSS ν¨λ”© κ¶μ¥)
        encrypted_key = recipient_public_key.encrypt(
            session_key,
            padding.OAEP(
                mgf=padding.MGF1(algorithm=hashes.SHA256()),
                algorithm=hashes.SHA256(),
                label=None
            )
        )

        # 3. AES-GCMμΌλ΅ λ°μ΄ν„° μ•”νΈν™”
        aesgcm = AESGCM(session_key)
        ciphertext = aesgcm.encrypt(nonce, plaintext.encode(), None)

        return {
            'encrypted_key': base64.b64encode(encrypted_key).decode(),
            'ciphertext': base64.b64encode(ciphertext).decode(),
            'nonce': base64.b64encode(nonce).decode()
        }

    def decrypt(self, encrypted_data: dict) -> str:
        """ν•μ΄λΈλ¦¬λ“ λ³µνΈν™”"""
        # 1. RSAλ΅ μ„Έμ… ν‚¤ λ³µνΈν™”
        encrypted_key = base64.b64decode(encrypted_data['encrypted_key'])
        session_key = self.private_key.decrypt(
            encrypted_key,
            padding.OAEP(
                mgf=padding.MGF1(algorithm=hashes.SHA256()),
                algorithm=hashes.SHA256(),
                label=None
            )
        )

        # 2. AESλ΅ λ°μ΄ν„° λ³µνΈν™”
        nonce = base64.b64decode(encrypted_data['nonce'])
        ciphertext = base64.b64decode(encrypted_data['ciphertext'])

        aesgcm = AESGCM(session_key)
        plaintext = aesgcm.decrypt(nonce, ciphertext, None)

        return plaintext.decode()

def sign_message(private_key, message: bytes) -> bytes:
    """RSA-PSS λ””μ§€ν„Έ μ„λ…"""
    signature = private_key.sign(
        message,
        sign_padding.PSS(
            mgf=sign_padding.MGF1(hashes.SHA256()),
            salt_length=sign_padding.PSS.MAX_LENGTH
        ),
        hashes.SHA256()
    )
    return signature

def verify_signature(public_key, message: bytes, signature: bytes) -> bool:
    """μ„λ… κ²€μ¦"""
    try:
        public_key.verify(
            signature, message,
            sign_padding.PSS(
                mgf=sign_padding.MGF1(hashes.SHA256()),
                salt_length=sign_padding.PSS.MAX_LENGTH
            ),
            hashes.SHA256()
        )
        return True
    except Exception:
        return False

const crypto = require('crypto');

// === AES-256-GCM μ•”νΈν™” ===
class AESCrypto {
    constructor(password) {
        // λΉ„λ°€λ²νΈμ—μ„ ν‚¤ νμƒ (scrypt μ‚¬μ©)
        this.salt = crypto.randomBytes(16);
        this.key = crypto.scryptSync(password, this.salt, 32); // 256λΉ„νΈ
    }

    encrypt(plaintext) {
        // 12λ°”μ΄νΈ IV (GCM κ¶μ¥)
        const iv = crypto.randomBytes(12);

        const cipher = crypto.createCipheriv('aes-256-gcm', this.key, iv);
        let encrypted = cipher.update(plaintext, 'utf8', 'base64');
        encrypted += cipher.final('base64');

        // Auth Tag (GCM λ¬΄κ²°μ„± κ²€μ¦μ©)
        const authTag = cipher.getAuthTag();

        return {
            ciphertext: encrypted,
            iv: iv.toString('base64'),
            authTag: authTag.toString('base64'),
            salt: this.salt.toString('base64')
        };
    }

    decrypt(encryptedData) {
        const decipher = crypto.createDecipheriv(
            'aes-256-gcm',
            this.key,
            Buffer.from(encryptedData.iv, 'base64')
        );

        decipher.setAuthTag(Buffer.from(encryptedData.authTag, 'base64'));

        let decrypted = decipher.update(encryptedData.ciphertext, 'base64', 'utf8');
        decrypted += decipher.final('utf8');

        return decrypted;
    }
}

function rsaEncrypt(publicKey, data) {
    return crypto.publicEncrypt(
        {
            key: publicKey,
            padding: crypto.constants.RSA_PKCS1_OAEP_PADDING,
            oaepHash: 'sha256'
        },
        Buffer.from(data)
    ).toString('base64');
}
